Natural Language Processing with Disaster Tweets¶

Predict which Tweets are about real disasters and which ones are not

https://www.kaggle.com/c/nlp-getting-started/

Train set consists of 7613 rows divided into 2 categories. There is test set with 3263 rows.

Columns¶

  • id - a unique identifier for each tweet
  • text - the text of the tweet
  • location - the location the tweet was sent from (may be blank)
  • keyword - a particular keyword from the tweet (may be blank)
  • target - this denotes whether a tweet is about a real disaster (1) or not (0)

I'll only use text and target column.

def emb_text_coverage(emb_dic, tokens):
    found = 0
    not_found = 0
    not_found_words = []
    for sentence in tokens:
        for w in sentence:
            if w in emb_dic:
                found += 1
            else:
                not_found_words.append(w)
                not_found +=1
    print('Text coverage: ' + str(round(found / (found + not_found) * 100, 2)))
    return not_found_words

def emb_vocab_coverage(emb_dic, tokens):
    found = 0
    not_found = 0
    tmp = []
    for x in tokens:
        for i in x:
            tmp.append(i)
    unique_words = list(set(tmp))
    for w in unique_words:
        if w in emb_dic:
                found += 1
        else:
            not_found +=1
    return 'Vocab coverage: ' + str(round(found / (found + not_found) * 100, 2))

def emb_coverage(emb_dic, tokens):
    lst = emb_text_coverage(emb_dic, tokens)
    print(emb_vocab_coverage(emb_dic, tokens))
    return Counter(lst)

def removeBadHashtags(text):
    lst = text.split()
    for i in range(len(lst) - 1, -1, -1):
        if('#' in lst[i]):
            lst.pop()
        else:
            break
    return ' '.join(lst)

        super(Mish, self).__init__(activation, **kwargs)
        self.__name__ = "Mish"


def mish(inputs):
    return inputs * tf.math.tanh(tf.math.softplus(inputs))

def recall_m(y_true, y_pred):
    true_positives = K.sum(K.round(K.clip(y_true * y_pred, 0, 1)))
    possible_positives = K.sum(K.round(K.clip(y_true, 0, 1)))
    recall = true_positives / (possible_positives + K.epsilon())
    return recall

def precision_m(y_true, y_pred):
    true_positives = K.sum(K.round(K.clip(y_true * y_pred, 0, 1)))
    predicted_positives = K.sum(K.round(K.clip(y_pred, 0, 1)))
    precision = true_positives / (predicted_positives + K.epsilon())
    return precision

def f1_m(y_true, y_pred):
    precision = precision_m(y_true, y_pred)
    recall = recall_m(y_true, y_pred)
    return 2*((precision*recall)/(precision+recall+K.epsilon()))

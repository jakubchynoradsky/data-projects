def cyclicalEncoding(dataframe, feature_name):
    """ Convert normal time or date feature to cyclical feature.
    :param dataframe: pandas dataframe
    :param feature_name: feature name
    """ 
    dataframe[feature_name + '_sin'] = np.sin(2 * np.pi * dataframe[feature_name] / dataframe[feature_name].max())
    dataframe[feature_name + '_cos'] = np.cos(2 * np.pi * dataframe[feature_name] / dataframe[feature_name].max())

def createLagsFromFeature(dataframe, feature_name):
    """ Creates new lagged(1-48, 72, 96, 120, 144, 168) features from any feature.
    :param dataframe: pandas dataframe
    :param feature_name: feature name
    """ 
    for i in range(1, 49):
        dataframe[feature_name + '_lag_' + str(i)] = dataframe[feature_name].shift(i)

def createRWFromFeature(dataframe, feature_name):
    """ Creates new rolling window(2-24) features from any feature.
    :param dataframe: pandas dataframe
    :param feature_name: feature name
    """
    for i in range(2, 25):
        dataframe[feature_name + '_roll_mean_' + str(i)] = dataframe[feature_name + '_lag_24'].rolling(window = i).mean()

def getHourLagsLabels(dataframe, hour):
    """ Extracts names of lag features from dataset based on hour.
    :param dataframe: pandas dataframe
    :param hour: hour of day
    :return: list with lag > hour
    """
    return [x for x in list(dataframe.columns) if 'lag' in x and int(x.split('_')[-1:][0]) > hour]

def createModel(number_of_features):    
    model = Sequential()    
    model.add(GRU(units = units, activation='tanh', return_sequences = True, input_shape = (1, number_of_features))) 
    model.add(Dropout(0.2))

    model.add(GRU(units = units,  activation='tanh', return_sequences = True))                                    
    model.add(Dropout(0.2))

    model.add(GRU(activation='tanh', units = units))
    model.add(Dropout(0.2))

    model.add(Dense(units=1, activation = "linear"))

    model.compile(loss = keras.losses.MeanSquaredError(), optimizer = keras.optimizers.Adam(learning_rate = learning_rate), metrics=['mae'])
    return model

def fitModel(model, batch_size, X_train, X_test, y_train, y_test):

    epochs = 200
    history = model.fit(X_train, y_train, epochs = epochs, batch_size = batch_size, validation_split = 0.2, shuffle = False, callbacks = [es])
    return model

def createAndFitModels(X_train, X_test, y_train, y_test, batch_size):
    """ Initialization and fit of models.
    :param X_train, X_test: pandas dataframes
    :param y_train, y_test: target features
    :param batch_size: model batch size
    :return: list with models and test sets for models
    """
    models = []
    test_sets = []
    for i in tqdm(range(forecast_horizon)):
        # Select only necessary features
        time_features = ['before_holiday', 'holiday', 'is_night', 'is_weekend', 'is_summer', 'is_year_transition', 'hour_sin', 'hour_cos', 'day_sin', 'day_cos', 'weekday_sin', 'weekday_cos', 'week_sin', 'week_cos', 'month_sin', 'month_cos']
        lag_features = getHourLagsLabels(X_train, i)
        roll_features = [x for x in list(X_train.columns) if 'roll' in x]
        expanding_features = [x for x in list(X_train.columns) if 'expanding_mean' in x]
        features = time_features + lag_features + roll_features + expanding_features
        
        # Create models
        model = createModel(len(features))
        
        X_train_features = X_train[features].copy()
        X_test_features = X_test[features].copy()
        X_train_features = np.array(X_train_features).reshape((X_train_features.shape[0], 1, X_train_features.shape[1]))
        X_test_features = np.array(X_test_features).reshape((X_test_features.shape[0], 1, X_test_features.shape[1]))
        
        # Fit models
        fitModel(model, batch_size, X_train_features, X_test_features, y_train, y_test)
        
        models.append(model)
        test_sets.append(X_test_features)
    return models, test_sets

def makePredictions(models, scaler, test_sets):
    """ Get models predictions.
    :param models: list with models
    :param scaler: target feature scaler
    :param test_sets: list with test sets for models
    :return: list with predictions
    """
    predictions = []
    for i in range(forecast_horizon):
        y_pred = models[i].predict(test_sets[i])
        y_pred = scaler.inverse_transform(y_pred).ravel()
        predictions.append(y_pred)
    return predictions
